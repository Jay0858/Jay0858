// Arithmetic Operators

void main() {

 int numone=50;
 int numtwo=20;
   
   
print('Addition is ${numone+numtwo}');
  
print('Subtraction is ${numone-numtwo}');
  
print('Multiply is ${numone*numtwo}');
  
print('Division is ${numone/numtwo}');
  
print('Modulus is ${numone%numtwo}');
  
}



output:
Addition is 70
Subtraction is 30
Multiply is 1000
Division is 2.5
Modulus is 10
